149    def get_relative_positions(self):
150        """Convert absolute block definition to position relative block
151        definition.
152        
153        Returns:
154            List of positions taken by the given block after translating
155            them relative to its base coordinates.
156        
157        """
158        positions = []
159
160        format = self.piece[self.rotation % len(self.piece)]
161
162        for i, line in enumerate(format):
163            row = line
164            for j, column in enumerate(row):
165                if column == '0':
166                    positions.append((self.x + j, self.y + i))
167
168        for i, pos in enumerate(positions):
169            positions[i] = (pos[0] - 2, pos[1] - 4)

171        return positions

174class Board:
175    """Representation of the game board.
176    
177    Attributes:
178        grid (list of lists of 3 element tuples): The board.
179        filled_positions (dictionary): Positions taken by blocks and their associated color.
180        next_block (boolean): Flag indicating whether to switch to next block.
181        current_block (Block): Block currently in motion.
182        nr_blocks (int): Number of blocks spawned.
183    
184    """
185
186    def __init__(self):
187        self.grid = []
188        self.filled_positions = {}
189        self.next_block = False
190        self.current_block = Block()
191        self.nr_blocks = 1
192
193    def zero_out_grid(self):
194        """Assign zero tuples to all positions."""
195        self.grid = [[(0,0,0) for x in range(10)] for x in range(20)]
196
197    def add_filled_positions_to_grid(self):
198        """Add filled_positions to grid."""
199        for i in range(len(self.grid)):
200            for j in range(len(self.grid[i])):
201                if (j,i) in self.filled_positions:
202                    block_color = self.filled_positions[(j,i)]
203                    self.grid[i][j] = block_color
204
205    def recreate_grid(self):
206        """Clear current grid and fill taken positions."""
207        self.zero_out_grid()
208        self.add_filled_positions_to_grid()
209
210    def is_valid(self):
211        """Check if current_block positions are valid.
212        
213        Returns:
214            True if all positions taken by the current_block are valid.
215            False if at least one field is not valid.
216        
217        """
218        valid_pos = [[(j, i) for j in range(10) if self.grid[i][j] == (0,0,0)] for i in range(20)]
219
220        valid_pos = [j for sub in valid_pos for j in sub]
221
222        block_pos = self.current_block.get_relative_positions()
223
224        for pos in block_pos:
225            if pos not in valid_pos:
226                if pos[1] > -1:
227                    return False
228        return True
229
230    def clear_any_filled_rows(self):
231        """Clear any filled rows."""
232        inc = 0
233        for i in range(len(self.grid)-1, -1, -1):
234            row = self.grid[i]
235            if (0, 0, 0) not in row:
236                inc += 1
237                ind = i
238                for j in range(len(row)):
239                    try:
240                        del self.filled_positions[(j, i)]
241                    except:
242                        continue
243
244        if inc > 0:
245            for key in sorted(list(self.filled_positions), key=lambda x: x[1])[::-1]:
246                x, y = key
247                if y < ind:
248                    newKey = (x, y + inc)
249                    self.filled_positions[newKey] = self.filled_positions.pop(key)
250
251    def is_game_over(self):
252         """Check if game over condition has been met.
253         
254         Returns:
255             True if game is over.
256             False if game is not over.
257         
258         """
259         for pos in self.filled_positions:
260             x, y = pos
261             if y < 1:
262                 return True
263         return False
264
265    def block_motion(self, fall_time):
266        """Move current_block downwards at a predefined rate.
267        
268        Returns:
269            int: The current fall_time if threshold has not been reached or zero
270                 if it has.
271        
272        """
273        fall_time_interval = 300
274
275        if fall_time >= fall_time_interval:
276            fall_time = 0
277            self.current_block.y += 1
278            if not self.is_valid() and self.current_block.y > 0:
279                self.current_block.y -= 1
280                self.next_block = True
281        return fall_time
282
283    def add_current_block_to_grid(self):
284        """Add current_block to grid."""
285        piece_pos = self.current_block.get_relative_positions()
286        for i in range(len(piece_pos)):
287            x, y = piece_pos[i]
288            if y > -1:
289                self.grid[y][x] = self.current_block.color
290
291    def end_current_block_motion(self):
292        """Shift focus to next block.
293        
294        Add positions taken by current_block to filled_positions.
295        Regenerate current_block.
296        
297        """
298        piece_pos = self.current_block.get_relative_positions()
299        if self.next_block:
300            for pos in piece_pos:
301                p = (pos[0], pos[1])
302                self.filled_positions[p] = self.current_block.color
303            self.current_block = Block()
304            self.nr_blocks += 1
305            self.next_block = False
306            self.clear_any_filled_rows()

327        """Draw text of specified size and color onto surface.

352            for j in range(col+1):
353                pygame.draw.line(self.surface, (128,128,128), (sx + j * 30, sy), (sx + j * 30, sy + BOARD_HEIGHT))  # vertical lines
354
355    def draw_window(self): 
356        """Draw window and contents."""
357        surface_color = pygame.colordict.THECOLORS.get("black")
358        self.surface.fill(surface_color)
359
360        for i in range(len(self.board.grid)):
361            for j in range(len(self.board.grid[i])):
362                pygame.draw.rect(self.surface, self.board.grid[i][j], (TOP_LEFT_X + j * 30, TOP_LEFT_Y + i * 30, 30, 30), 0)

364        self.draw_grid(20, 10)

367class PythonBlockDrop:
368    """Tetris game."""
369
370    def __init__(self):
371        pygame.font.init()
372        self.surface = pygame.display.set_mode((WINDOW_WIDTH, WINDOW_HEIGHT))
373        pygame.display.set_caption('Tetris')
374        icon = pygame.image.load('icon.png')
375        pygame.display.set_icon(icon)
376        self.board = Board()
377        self.boardview = BoardView(self.surface, self.board)
378        self.menu_loop()
379
380    def game_event_handler(self, run):
381        """Handle in game events.
382        
383        Args:
384            run (boolean): Loop control flag.
385
386        """
387        for event in pygame.event.get():
388            if event.type == pygame.QUIT:
389                run = False
390                pygame.display.quit()
391                quit()
392
393            if event.type == pygame.KEYDOWN:
394                if event.key == pygame.K_LEFT:
395                    self.board.current_block.x -= 1
396                    if not self.board.is_valid():
397                        self.board.current_block.x += 1
398                elif event.key == pygame.K_RIGHT:
399                    self.board.current_block.x += 1
400                    if not self.board.is_valid():
401                        self.board.current_block.x -= 1
402                elif event.key == pygame.K_UP:
403                    self.board.current_block.rotation = self.board.current_block.rotation + 1 % len(self.board.current_block.piece)
404                    if not self.board.is_valid():
405                        self.board.current_block.rotation = self.board.current_block.rotation - 1 % len(self.board.current_block.piece)
406                elif event.key == pygame.K_DOWN:
407                    self.board.current_block.y += 2
408                    if not self.board.is_valid():
409                        self.board.current_block.y -= 2
410        return run
411
412    def game_loop(self):
413        """PythonBlockDrop game loop."""
414        run = True
415        clock = pygame.time.Clock()
416        fall_time = 0
417
418        while run:
419            fall_time += clock.get_rawtime()
420            clock.tick()
421
422            # schedule changes by shifting them to grid
423
424            self.board.recreate_grid()
425            fall_time = self.board.block_motion(fall_time)
426            run = self.game_event_handler(run)
427            self.board.add_current_block_to_grid()
428            self.board.end_current_block_motion()
429
430            # push changes applied to grid to display
431
432            self.boardview.draw_window()
433            pygame.display.update()
434            run = not self.board.is_game_over()
435        self.game_over_loop()
436
437    def game_over_loop(self):
438        """ PythonBlockDrop game over loop."""
439        run = True
440        while run:
441            self.surface.fill((0,0,0))
442            text_position = (TOP_LEFT_X + BOARD_WIDTH / 2,
443                             TOP_LEFT_Y + BOARD_HEIGHT / 8)
444            self.boardview.draw_text("Game Over", 60, (255,255,255), text_position)
445            text_position = (TOP_LEFT_X + BOARD_WIDTH / 2,
446                             TOP_LEFT_Y + 3 * BOARD_HEIGHT / 8)
447            self.boardview.draw_text('Your score is:', 40, (255, 255, 255), text_position)
448            text_position = (TOP_LEFT_X + BOARD_WIDTH / 2,
449                             TOP_LEFT_Y + BOARD_HEIGHT / 2)
450            self.boardview.draw_text(str(self.board.nr_blocks)+" blocks", 40, (255, 255, 255), text_position)
451            text_position = (TOP_LEFT_X + BOARD_WIDTH / 2,
452                             TOP_LEFT_Y + 6 * BOARD_HEIGHT / 8)
453            self.boardview.draw_text("Press any key to quit.", 50, (255, 255, 255), text_position)
454            pygame.display.update()
455
456            for event in pygame.event.get():
457                if event.type == pygame.QUIT:
458                    run = False
459                if event.type == pygame.KEYDOWN:
460                    run = False
461        pygame.quit()
462            
463    def menu_loop(self):
464        """PythonBlockDrop menu loop."""
465        run = True
466        while run:
467            self.surface.fill((0,0,0))
468            text_position = (TOP_LEFT_X + BOARD_WIDTH / 2,
469                             TOP_LEFT_Y + BOARD_HEIGHT / 2)
470            self.boardview.draw_text('Press any key to begin.', 60, (255, 255, 255), text_position)
471            pygame.display.update()
472            
473            for event in pygame.event.get():
474                if event.type == pygame.QUIT:
475                    run = False
476                if event.type == pygame.KEYDOWN:
477                    self.game_loop()
478                    run = False
479        pygame.quit()
